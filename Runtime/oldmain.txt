#undef main
int main(int argc, const char** argv)
{
#ifdef SERVER
	client = false;
#endif
	Hyper::GameEngine engine(client);

	ZombieSharedLayer* shared = new ZombieSharedLayer();
	engine.GetLayerStack().PushLayer(shared);

	if (client)
	{
#ifndef SERVER
		engine.GetLayerStack().PushOverlay(new ZombieGameLayer(*shared));
#endif
	}
	else
	{
#ifdef SERVER
		engine.GetLayerStack().PushOverlay(new ZombieServerLayer(*shared));
#endif
	}

	engine.Start();

	return 0;
}
